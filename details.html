<script>

  const tableBody = document.querySelector('#medicineTable tbody');

  const form = document.getElementById('addMedicineForm');


  let medicines = JSON.parse(localStorage.getItem('medicines')) || [];


  function renderTable() {

    tableBody.innerHTML = '';

    medicines.forEach((med, index) => {

      const remaining = med.initial - med.sold;

      const status = remaining <= med.reorder ? "LOW STOCK" : "OK";

      const statusClass = status === "LOW STOCK" ? "low" : "ok";


      const row = document.createElement('tr');

      row.innerHTML = `

        <td>${index + 1}</td>

        <td>${med.name}</td>

        <td>${med.initial}</td>

        <td>${med.sold}</td>

        <td>${remaining}</td>

        <td>${med.reorder}</td>

        <td class="${statusClass}">${status}</td>

        <td>${med.lastBought || '-'}</td>

      `;

      tableBody.appendChild(row);

    });

  }


  form.addEventListener('submit', e => {

    e.preventDefault();


    const name = document.getElementById('name').value.trim();

    const initial = parseInt(document.getElementById('initial').value);

    const sold = parseInt(document.getElementById('sold').value);

    const reorder = parseInt(document.getElementById('reorder').value);

    const lastBought = document.getElementById('lastBought').value;


    // Check for duplicate medicine names (case-insensitive)

    if (medicines.some(med => med.name.toLowerCase() === name.toLowerCase())) {

      alert('Medicine already exists!');

      return;

    }


    medicines.push({

      name,

      initial,

      sold,

      reorder,

      lastBought

    });


    localStorage.setItem('medicines', JSON.stringify(medicines));


    form.reset();

    renderTable();

  });


  // Initial render

  renderTable();

</script>
